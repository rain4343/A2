@push('scripts')
<script>
    function dragMoveListener(event) {
        var target = event.target;
        var x = (parseFloat(target.getAttribute('data-x')) || 0) + event.dx;
        var y = (parseFloat(target.getAttribute('data-y')) || 0) + event.dy;

        target.style.transform = 'translate(' + x + 'px, ' + y + 'px)';
        target.setAttribute('data-x', x);
        target.setAttribute('data-y', y);
    }

    interact('.draggable').draggable({
        listeners: { move: dragMoveListener },
        inertia: true,
        modifiers: [
            interact.modifiers.restrictRect({
                restriction: 'parent',
                endOnly: true
            })
        ]
    }).resizable({
        edges: { left: true, right: true, bottom: true, top: true },
        listeners: {
            move (event) {
                var target = event.target
                var x = (parseFloat(target.getAttribute('data-x')) || 0)
                var y = (parseFloat(target.getAttribute('data-y')) || 0)

                target.style.width = event.rect.width + 'px'
                target.style.height = event.rect.height + 'px'

                x += event.deltaRect.left
                y += event.deltaRect.top

                target.style.transform = 'translate(' + x + 'px,' + y + 'px)'
                target.setAttribute('data-x', x)
                target.setAttribute('data-y', y)
            }
        },
        modifiers: [
            interact.modifiers.restrictSize({
                min: { width: 100, height: 50 }
            })
        ],
        inertia: true
    });

    document.getElementById('signature-form').addEventListener('submit', function(e) {
        // Signature
        const sigBox = document.getElementById('signature-box');
        if (sigBox) {
            document.getElementById('signature_x').value = sigBox.offsetLeft + (parseFloat(sigBox.getAttribute('data-x')) || 0);
            document.getElementById('signature_y').value = sigBox.offsetTop + (parseFloat(sigBox.getAttribute('data-y')) || 0);
        }

        // Notes
        const notesBox = document.getElementById('notes-box');
        document.getElementById('notes_x').value = notesBox.offsetLeft + (parseFloat(notesBox.getAttribute('data-x')) || 0);
        document.getElementById('notes_y').value = notesBox.offsetTop + (parseFloat(notesBox.getAttribute('data-y')) || 0);
        document.getElementById('notes_width').value = notesBox.offsetWidth;
        document.getElementById('notes_height').value = notesBox.offsetHeight;
        document.getElementById('notes_text_hidden').value = document.getElementById('notes-text').value;

        // Date
        const dateBox = document.getElementById('date-box');
        document.getElementById('date_x').value = dateBox.offsetLeft + (parseFloat(dateBox.getAttribute('data-x')) || 0);
        document.getElementById('date_y').value = dateBox.offsetTop + (parseFloat(dateBox.getAttribute('data-y')) || 0);
    });
</script>
